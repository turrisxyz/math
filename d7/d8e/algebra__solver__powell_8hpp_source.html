55  Eigen::HybridNonLinearSolver<hybrj_functor_solver<F>> solver(hfs);
56 
57  // Compute theta_dbl

59  solver.parameters.maxfev = max_num_steps;
60  solver.solve(x);
61 
62  // Check if the max number of steps has been exceeded
63  if (solver.nfev >= max_num_steps) {
64  [max_num_steps]() STAN_COLD_PATH {
65  throw_domain_error("algebra_solver", "maximum number of iterations",
66  max_num_steps, "(", ") was exceeded in the solve.");
67  }();
68  }
69 
70  // Check solution is a root

79  function_tolerance, "",
80  ". Consider decreasing the relative tolerance and "
81  "increasing max_num_steps.");
