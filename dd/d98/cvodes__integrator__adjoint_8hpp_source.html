400  void chain() final {
401  if (!is_var_return_) {
402  return;
403  }
404 
405  // for sensitivities wrt to ts we do not need to run the backward
406  // integration
407  if (is_var_ts_) {
408  Eigen::VectorXd step_sens = Eigen::VectorXd::Zero(N_);
409  for (int i = 0; i < solver_->ts_.size(); ++i) {
410  for (int j = 0; j < N_; ++j) {
411  step_sens.coeffRef(j) += y_return_varis_[N_ * i + j]->adj_;
412  }
413 
414  adjoint_of(solver_->ts_[i])
415  += step_sens.dot(rhs(value_of(solver_->ts_[i]), solver_->y_[i],
416  solver_->value_of_args_tuple_));
417  step_sens.setZero();
418  }
419 
420  if (is_var_only_ts_) {
421  return;
422  }
423  }
424 
425  solver_->state_backward_.setZero();
426  solver_->quad_.setZero();
427 
428  // At every time step, collect the adjoints from the output
429  // variables and re-initialize the solver
430  double t_init = value_of(solver_->ts_.back());
431  for (int i = solver_->ts_.size() - 1; i >= 0; --i) {
432  // Take in the adjoints from all the output variables at this point
433  // in time
434  for (int j = 0; j < N_; ++j) {
435  solver_->state_backward_.coeffRef(j)
436  += y_return_varis_[N_ * i + j]->adj_;
437  }
438 
439  double t_final = value_of((i > 0) ? solver_->ts_[i - 1] : solver_->t0_);
440  if (t_final != t_init) {
441  if (unlikely(!backward_is_initialized_)) {
442  CHECK_CVODES_CALL(CVodeCreateB(solver_->cvodes_mem_, solver_backward_,
443  &index_backward_));
444 
445  CHECK_CVODES_CALL(CVodeSetUserDataB(solver_->cvodes_mem_,
446  index_backward_,
447  reinterpret_cast<void*>(this)));
448 
449  // initialize CVODES backward machinery.
450  // the states of the backward problem *are* the adjoints
451  // of the ode states
452  CHECK_CVODES_CALL(
453  CVodeInitB(solver_->cvodes_mem_, index_backward_,
454  &cvodes_integrator_adjoint_vari::cv_rhs_adj, t_init,
455  solver_->nv_state_backward_));
456 
457  CHECK_CVODES_CALL(
458  CVodeSVtolerancesB(solver_->cvodes_mem_, index_backward_,
459  relative_tolerance_backward_,
460  solver_->nv_absolute_tolerance_backward_));
461 
462  CHECK_CVODES_CALL(CVodeSetMaxNumStepsB(
463  solver_->cvodes_mem_, index_backward_, max_num_steps_));
464 
465  CHECK_CVODES_CALL(CVodeSetLinearSolverB(
466  solver_->cvodes_mem_, index_backward_, solver_->LS_backward_,
467  solver_->A_backward_));
468 
469  CHECK_CVODES_CALL(CVodeSetJacFnB(
470  solver_->cvodes_mem_, index_backward_,
471  &cvodes_integrator_adjoint_vari::cv_jacobian_rhs_adj_states));
472 
473  // Allocate space for backwards quadrature needed when
474  // parameters vary.
475  if (is_any_var_args_) {
476  CHECK_CVODES_CALL(
477  CVodeQuadInitB(solver_->cvodes_mem_, index_backward_,
478  &cvodes_integrator_adjoint_vari::cv_quad_rhs_adj,
479  solver_->nv_quad_));
480 
481  CHECK_CVODES_CALL(
482  CVodeQuadSStolerancesB(solver_->cvodes_mem_, index_backward_,
483  relative_tolerance_quadrature_,
484  absolute_tolerance_quadrature_));
485 
486  CHECK_CVODES_CALL(CVodeSetQuadErrConB(solver_->cvodes_mem_,
487  index_backward_, SUNTRUE));
488  }
489 
490  backward_is_initialized_ = true;
491  } else {
492  // just re-initialize the solver
493 
494  CHECK_CVODES_CALL(CVodeReInitB(solver_->cvodes_mem_, index_backward_,
495  t_init, solver_->nv_state_backward_));
496 
497  if (is_any_var_args_) {
498  CHECK_CVODES_CALL(CVodeQuadReInitB(
499  solver_->cvodes_mem_, index_backward_, solver_->nv_quad_));
500  }
501  }
502 
503  CHECK_CVODES_CALL(CVodeB(solver_->cvodes_mem_, t_final, CV_NORMAL));
504 
505  // obtain adjoint states and update t_init to time point
506  // reached of t_final
507  CHECK_CVODES_CALL(CVodeGetB(solver_->cvodes_mem_, index_backward_,
508  &t_init, solver_->nv_state_backward_));
509 
510  if (is_any_var_args_) {
511  CHECK_CVODES_CALL(CVodeGetQuadB(solver_->cvodes_mem_, index_backward_,
512  &t_init, solver_->nv_quad_));
513  }
514  }
515  }
516 
517  // After integrating all the way back to t0, we finally have the
518  // the adjoints we wanted
519 
520  // This is the dlog_density / d(initial_time_point) adjoint
521  if (is_var_t0_) {
522  adjoint_of(solver_->t0_) += -solver_->state_backward_.dot(
523  rhs(t_init, value_of(solver_->y0_), solver_->value_of_args_tuple_));
524  }
525 
526  // These are the dlog_density / d(initial_conditions[s]) adjoints
527  if (is_var_y0_t0_) {
528  forward_as<Eigen::Matrix<var, Eigen::Dynamic, 1>>(solver_->y0_).adj()
529  += solver_->state_backward_;
530  }
531 
532  // These are the dlog_density / d(parameters[s]) adjoints
533  if (is_any_var_args_) {
534  for (size_t s = 0; s < num_args_vars_; ++s) {
535  args_varis_[s]->adj_ += solver_->quad_.coeff(s);
536  }
537  }
538  }

669

676  fy_var.size(), "states", N_);

679  Jfy.col(0) = y_var.adj();

681  nested.set_zero_all_adjoints();
682  grad(fy_var.coeffRef(i).vi_);
683  Jfy.col(i) = y_var.adj();
684  }
