217  int get_twisted_factorization(
218  const __global double_d* l, const __global double_d* d, double_d shift,
219  __global double_d* l_plus, __global double_d* u_minus,
220  __global double_d* s) {

222  int gid = get_global_id(0);
223  int m = n - 1;
224  // calculate shifted ldl
225  s[gid] = neg(shift);
226  for (int i = 0; i < m; i++) {
227  double_d d_plus = add_dd_dd(s[i * n + gid], d[i * n + gid]);
228  l_plus[i * n + gid]
229  = mul_dd_dd(l[i * n + gid], div_dd_dd(d[i * n + gid], d_plus));
230  if (isnan_dd(l_plus[i * n + gid])) { // d_plus==0
231  // one (or both) of d[i], l[i] is very close to 0
232  if (lt_dd_dd(abs_dd(l[i * n + gid]), abs_dd(d[i * n + gid]))) {
233  l_plus[i * n + gid]
234  = mul_dd_d(d[i * n + gid], copysign_d_dd(1., l[i * n + gid])
235  * copysign_d_dd(1., d_plus));
236  } else {
237  l_plus[i * n + gid]
238  = mul_dd_d(l[i * n + gid], copysign_d_dd(1., d[i * n + gid])
239  * copysign_d_dd(1., d_plus));
240  }
241  }
242  s[(i + 1) * n + gid] = sub_dd_dd(
243  mul_dd_dd(mul_dd_dd(l_plus[i * n + gid], l[i * n + gid]),
244  s[i * n + gid]),
245  shift);
246  if (isnan_dd(s[(i + 1) * n + gid])) {
247  if (gt_dd_dd(abs_dd(l_plus[i * n + gid]),
248  abs_dd(s[i * n + gid]))) { // l_plus[i * n + gid] == inf
249  if (gt_dd_dd(abs_dd(s[i * n + gid]),
250  abs_dd(l[i * n + gid]))) { // l[i*n+gid]==0
251  s[(i + 1) * n + gid] = sub_dd_dd(
252  mul_dd_d(s[i * n + gid],
253  copysign_d_dd(1., l[i * n + gid])
254  * copysign_d_dd(1., l_plus[i * n + gid])),
255  shift);
256  } else { // s[i*n+gid]==0
257  s[(i + 1) * n + gid] = sub_dd_dd(
258  mul_dd_d(l[i * n + gid],
259  copysign_d_dd(1., s[i * n + gid])
260  * copysign_d_dd(1., l_plus[i * n + gid])),
261  shift);
262  }
263  } else { // s[i*n+gid]==inf
264  if (gt_dd_dd(abs_dd(l_plus[i * n + gid]),
265  abs_dd(l[i * n + gid]))) { // l[i]==0
266  s[(i + 1) * n + gid]
267  = sub_dd_dd(mul_dd_d(l_plus[i * n + gid],
268  copysign_d_dd(1., l[i * n + gid])
269  * copysign_d_dd(1., s[i * n + gid])),
270  shift);
271  } else { // l_plus[i]==0
272  s[(i + 1) * n + gid] = sub_dd_dd(
273  mul_dd_d(l[i * n + gid],
274  copysign_d_dd(1., s[i * n + gid])
275  * copysign_d_dd(1., l_plus[i * n + gid])),
276  shift);
277  }
278  }
279  }
280  }
281  // calculate shifted udu and twist index
282  double_d p = sub_dd_dd(d[m * n + gid], shift);
283  double_d min_gamma = abs_dd(add_dd_dd(s[m * n + gid], d[m * n + gid]));
284  int twist_index = m;
285 
286  for (int i = m - 1; i >= 0; i--) {
287  double_d d_minus
288  = add_dd_dd(mul_dd_dd(mul_dd_dd(d[i * n + gid], l[i * n + gid]),
289  l[i * n + gid]),
290  p);
291  double_d t = div_dd_dd(d[i * n + gid], d_minus);
292  u_minus[i * n + gid] = mul_dd_dd(l[i * n + gid], t);
293  if (isnan_dd(u_minus[i * n + gid])) {
294  if (isnan_dd(t)) {
295  double t_high = copysign_d_dd(1., d[i * n + gid])
296  * copysign_d_dd(1., d_minus);
297  t.high = t_high;
298  t.low = 0;
299  u_minus[i * n + gid] = mul_dd_d(l[i * n + gid], t_high);
300  } else { // t==inf, l[i*n+gid]==0
301  u_minus[i * n + gid]
302  = mul_dd_d(d[i * n + gid], copysign_d_dd(1., l[i * n + gid])
303  * copysign_d_dd(1., t));
304  }
305  }
306  double_d gamma = abs_dd(add_dd_dd(s[i * n + gid], mul_dd_dd(t, p)));
307  if (isnan_dd(gamma)) { // t==inf, p==0 OR t==0, p==inf
308  double_d d_sign
309  = mul_dd_d(d[i * n + gid],
310  copysign_d_dd(1., d_minus) * copysign_d_dd(1., t));
311  gamma = abs_dd(add_dd_dd(s[i * n + gid], d_sign));
312  p = sub_dd_dd(d_sign, shift);
313  } else { // usual case
314  p = sub_dd_dd(mul_dd_dd(p, t), shift);
315  }
316  if (lt_dd_dd(gamma, min_gamma)) {
317  min_gamma = gamma;
318  twist_index = i;
319  }
320  }
321  return twist_index;
322  }

333  void calculate_eigenvector(const __global double_d* l_plus,
334  const __global double_d* u_minus,
335  const __global double* subdiag, int twist_idx,
336  __global double* eigenvectors) {
337  int n = get_global_size(0);
338  int gid = get_global_id(0);
339  int i = gid;
340  eigenvectors[twist_idx * n + gid] = 1;
341  double_d last = (double_d){1, 0};
342  double_d last2 = (double_d){1, 0};
343  double norm = 1;
344  // part of the eigenvector after the twist index
345  for (int j = twist_idx + 1; j < n; j++) {
346  if (last.high != 0 || last.low != 0) {
347  last2 = last;
348  last = neg(mul_dd_dd(u_minus[(j - 1) * n + gid], last));
349  eigenvectors[j * n + gid] = last.high;
350  } else {
351  double_d tmp = last;
352  last = mul_dd_d(last2, -subdiag[j - 2] / subdiag[j - 1]);
353  last2 = tmp;
354  if (isnan(last.high) || isinf(last.high)) { // subdiag[j - 1]==0
355  last = (double_d){0, 0};
356  }
357  eigenvectors[j * n + gid] = last.high;
358  }
359  norm += eigenvectors[j * n + gid] * eigenvectors[j * n + gid];
360  }
361  last = (double_d){eigenvectors[twist_idx * n + gid], 0};
362  last2 = (double_d){1, 0};
363  // part of the eigenvector before the twist index
364  for (int j = twist_idx - 1; j >= 0; j--) {
365  if (last.high != 0 || last.low != 0) {
366  last2 = last;
367  last = neg(mul_dd_dd(l_plus[j * n + gid], last));
368  eigenvectors[j * n + gid] = last.high;
369  } else {
370  double_d tmp = last;
371  last = mul_dd_d(last2, -subdiag[j + 1] / subdiag[j]);
372  if (isnan(last.high) || isinf(last.high)) { // subdiag[j]==0
373  last = (double_d){0, 0};
374  }
375  eigenvectors[j * n + gid] = last.high;
376  }
377  norm += eigenvectors[j * n + gid] * eigenvectors[j * n + gid];
378  }
379  norm = 1 / sqrt(norm);
380  // normalize the eigenvector
381  for (int j = 0; j < n; j++) {
382  eigenvectors[j * n + gid] *= norm;
383  }
384  }
